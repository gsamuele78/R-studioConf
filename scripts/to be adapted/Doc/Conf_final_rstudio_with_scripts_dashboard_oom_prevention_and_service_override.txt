# =============================================================
# BIOME-CALC SYSTEM PROFILE v5.7 (Omni-Shield Consolidated)
# =============================================================
# - Protezione Cgroups v2 + Rlimit (Safe RAM)
# - Trasparenza Parquet (Rete 1Gbps) per read.csv e fread
# - Notifiche Post-Mortem + Notifiche Cambio Quota Live
# - Integrazione rgee, Keras, TensorFlow & Terra (RAMDisk)
# =============================================================

local({
  # --- 1. PARAMETRI HARDWARE & QUOTE ---
  mem_total <- as.numeric(system("awk '/MemTotal/ {print $2}' /proc/meminfo", intern = TRUE))
  ram_host_gb <- floor(mem_total / 1024 / 1024)
  riserva_os_tmpfs <- 115 # 100GB RAMDisk + 15GB OS/Buffer
  ram_per_r <- ram_host_gb - riserva_os_tmpfs
  total_cores  <- parallel::detectCores()
  
  .last_quota <- 0 # Stato per notifiche live

  update_resources <- function(quiet = TRUE) {
    n_procs <- as.integer(system("pgrep -c '^R$|^rsession$'", intern = TRUE))
    if (is.na(n_procs) || n_procs <= 0) n_procs <- 1
    
    # Calcolo Fair: RAM (90% quota) e Core (Totale - 2 per OS)
    quota_ram <- floor((ram_per_r / n_procs) * 0.9)
    fair_cores <- floor((total_cores - 2) / n_procs)
    if (fair_cores < 1) fair_cores <- 1

    # Applicazione Limiti RAM (unix rlimit)
    if (requireNamespace("unix", quietly = TRUE)) {
      try(unix::rlimit_as(quota_ram * 1e9), silent = TRUE)
    }

    # Parallelizzazione Automatica (terra, future)
    if (requireNamespace("future", quietly = TRUE)) {
      future::plan(future::multisession, workers = fair_cores)
    }
    if (requireNamespace("terra", quietly = TRUE)) {
      terra::terraOptions(cores = fair_cores, tempdir = "/tmp", memfrac = 0.5)
    }

    # --- NOTIFICHE DINAMICHE ---
    if (interactive()) {
      if (!quiet) {
        # Messaggio di Benvenuto (Al Login)
        cat("\n", rep("=", 55), "\n", sep="")
        cat("🌱 BIOME-CALC v5.7 | Omni-Shield Architecture\n")
        cat(rep("-", 55), "\n", sep="")
        cat(sprintf("👥 Utenti Online:   %d\n", n_procs))
        cat(sprintf("📊 Quota RAM:       %d GB | RAMDisk: 100 GB\n", quota_ram))
        cat(sprintf("🚀 Core Assegnati:   %d (su %d totali)\n", fair_cores, total_cores))
        cat(rep("-", 55), "\n", sep="")
        cat("💡 Ottimizzazione Rete 1Gbps: Parquet Transparency [ON]\n")
        cat(rep("=", 55), "\n\n", sep="")
      } else if (.last_quota > 0 && quota_ram < .last_quota) {
        message(sprintf("\n🔔 BIOME-INFO: Nuovo utente rilevato. Quota RAM aggiornata: %d GB", quota_ram))
      }
    }
    
    .last_quota <<- quota_ram
    gc(full = TRUE, verbose = FALSE)
  }

  update_resources(quiet = FALSE)
  addTaskCallback(function(...) { update_resources(quiet = TRUE); TRUE })

  # --- 2. WRAPPER TRASPARENTE PARQUET (No library required) ---
  if (requireNamespace("arrow", quietly = TRUE)) {
    biome_smart_loader <- function(file, ...) {
      if (is.character(file) && length(file) == 1) {
        pq_path <- paste0(normalizePath(file, mustWork = FALSE), ".parquet")
        if (file.exists(pq_path)) {
          if (interactive()) message(sprintf("🚀 BIOME-I/O: Caricamento Parquet [%s]", basename(file)))
          res <- try(arrow::read_parquet(pq_path), silent = TRUE)
          if (!inherits(res, "try-error")) return(res)
        }
      }
      # Fallback al Namespace originale per evitare loop
      call_func <- as.character(sys.call())
      if (grepl("fread", call_func) && requireNamespace("data.table", quietly = TRUE)) {
        return(asNamespace("data.table")$fread(file, ...))
      } else {
        return(asNamespace("utils")$read.csv(file, ...))
      }
    }

    # Override utils::read.csv
    utils_env <- as.environment("package:utils")
    unlockBinding("read.csv", utils_env); assign("read.csv", biome_smart_loader, envir = utils_env); lockBinding("read.csv", utils_env)
    
    # Override data.table::fread (Lazy Loading)
    setHook(packageEvent("data.table", "onLoad"), function(...) {
      dt_env <- as.environment("package:data.table")
      unlockBinding("fread", dt_env); assign("fread", biome_smart_loader, envir = dt_env); lockBinding("fread", dt_env)
    })
  }

  # --- 3. IA & CLOUD (Keras, TF, rgee) ---
  py_path <- "/opt/r-geospatial/bin/python"
  if (file.exists(py_path)) {
    Sys.setenv(RETICULATE_PYTHON = py_path, EARTHENGINE_PYTHON = py_path)
  }
  Sys.setenv(TF_FORCE_GPU_ALLOW_GROWTH = "true")
  Sys.setenv(TF_CPP_MIN_LOG_LEVEL = "2")

  # --- 4. FUNZIONE STATO MANUALE ---
  biome_status <<- function() update_resources(quiet = FALSE)

  options(stringsAsFactors = FALSE, timeout = 300)
})

3. Monitoraggio Attivo: /usr/local/bin/monitor_biome.sh

Questo script killa i processi che superano la soglia di sicurezza prima che l'intero sistema vada in OOM.
bash
#!/bin/bash
# Monitoraggio Risorse BIOME-CALC: Prevenzione OOM e Notifiche
THRESHOLD_RAM_GB=8
LOG="/var/log/biome_monitor.log"

while true; do
    FREE_RAM=$(free -g | awk '/^Mem:/{print $4}')
    
    if [ "$FREE_RAM" -lt "$THRESHOLD_RAM_GB" ]; then
        # Identifica il processo R più pesante
        PID_INFO=$(ps -eo pid,user,%mem,args --sort=-%mem | grep -E "rsession|R$" | grep -v grep | head -n 1)
        PID=$(echo $PID_INFO | awk '{print $1}')
        USER_NAME=$(echo $PID_INFO | awk '{print $2}')
        SCRIPT_NAME=$(echo $PID_INFO | cut -d' ' -f4-)
        
        if [ ! -z "$PID" ]; then
            # Genera file di crash per l'utente (Rprofile lo leggerà al login)
            { 
              echo "DATA: $(date '+%d/%m %H:%M')"
              echo "PID: $PID"
              echo "COMANDO: $SCRIPT_NAME"
              echo "MOTIVO: RAM libera < ${THRESHOLD_RAM_GB}GB (Emergenza Cluster)"
            } > "/home/$USER_NAME/ULTIMO_CRASH_RAM.txt"
            
            chown $USER_NAME "/home/$USER_NAME/ULTIMO_CRASH_RAM.txt"
            echo "$(date): KILL PID $PID ($USER_NAME) - $SCRIPT_NAME" >> $LOG
            kill -9 $PID
        fi
    fi
    sleep 60
done
